1//! Wavelet Matrix に、ビットごとのSegment Treeを追加することで、
2//! 二次元セグ木と同様に(オフラインな)1点更新、矩形和を求められる  
3//! 現状座標圧縮二次元セグ木よりも高速
4
5use algebra::{Commutative, Group, Monoid};
6use bitdict::BitDict;
7use internal_bits::ceil_log2;
8use internal_type_traits::Integral;
9use segtree::SegTree;
10use std::ops::RangeBounds;

12/// 座標圧縮とx座標の重複除去を行うWrapper Tが座標圧縮する型  
13/// 可換なモノイドのオフラインな1点更新、二次元矩形区間和クエリに対応
14pub struct WMSegWrapper<M: Monoid + Commutative, T: Integral> {
15    wm: WaveletMatrixSegTree<M>,
16    sorted_y: Vec<T>,
17    x_y: Vec<(T, T)>,
18}
19
20impl<M: Monoid + Commutative, T: Integral> WMSegWrapper<M, T> {
21    /// すべて単位元で初期化する場合
22    pub fn new(update_points: Vec<(T, T)>) -> Self {
23        Self::from_weight(update_points, &[])
24    }
25
26    /// update_pointsは更新クエリのある点の座標のリスト ただしinit_weightsの点も含める  
27    /// init_weightsは初期状態の点の座標と重みのリスト (x, y, w)  
28    /// もしinit_weightsの点が重複する場合は、それらmonoidの積として初期化するので注意(上書きしたい場合は事前に重複を消す前処理をしてください)
29    pub fn from_weight(mut update_points: Vec<(T, T)>, init_weights: &[(T, T, M::Target)]) -> Self {
30        update_points.sort_unstable();
31        update_points.dedup();
32        let mut sorted_y = update_points
33            .iter()
34            .map(|(_, y)| y)
35            .copied()
36            .collect::<Vec<_>>();
37        sorted_y.sort_unstable();
38        let compressed_list = update_points
39            .iter()
40            .map(|(_, y)| sorted_y.binary_search(y).unwrap())
41            .collect::<Vec<_>>();
42        let mut weight_list = vec![M::id_element(); update_points.len()];
43        for (x, y, w) in init_weights {
44            let idx = update_points
45                .binary_search(&(*x, *y))
46                .expect("init_weight points are not in update_points!!!");
47            weight_list[idx] = M::binary_operation(&weight_list[idx], w);
48        }
49        let wm = WaveletMatrixSegTree::<M>::from_weight(&compressed_list, &weight_list);
50        Self {
51            wm,
52            sorted_y,
53            x_y: update_points,
54        }
55    }
56
57    fn get_pos_range<R: RangeBounds<T>>(&self, range: R) -> (usize, usize) {
58        use std::ops::Bound::*;
59        let l = match range.start_bound() {
60            Included(&l) => l,
61            Excluded(&l) => l + T::one(),
62            Unbounded => T::min_value(),
63        };
64        let r = match range.end_bound() {
65            Included(&r) => r + T::one(),
66            Excluded(&r) => r,
67            Unbounded => T::max_value(),
68        };
69        assert!(l <= r);
70        let l = self.x_y.partition_point(|&(x, _)| x < l);
71        let r = self.x_y.partition_point(|&(x, _)| x < r);
72        (l, r)
73    }
74
75    fn get_num_range<R: RangeBounds<T>>(&self, range: R) -> (usize, usize) {
76        use std::ops::Bound::*;
77        let l = match range.start_bound() {
78            Included(&l) => l,
79            Excluded(&l) => l + T::one(),
80            Unbounded => T::min_value(),
81        };
82        let r = match range.end_bound() {
83            Included(&r) => r + T::one(),
84            Excluded(&r) => r,
85            Unbounded => T::max_value(),
86        };
87        assert!(l <= r);
88        let l = self.sorted_y.partition_point(|&y| y < l);
89        let r = self.sorted_y.partition_point(|&y| y < r);
90        (l, r)
91    }
92
93    /// 点(x, y)の重みをnew_valに更新する
94    pub fn set(&mut self, x: T, y: T, new_val: M::Target) {
95        let x = self
96            .x_y
97            .binary_search(&(x, y))
98            .expect("(x, y) is not in update_queries!!!");
99        self.wm.set(x, new_val);
100    }
101
102    /// 点(x, y)の重みを取得する
103    pub fn get(&self, x: T, y: T) -> M::Target {
104        let Ok(x) = self.x_y.binary_search(&(x, y)) else {
105            return M::id_element();
106        };
107        self.wm.get_weight(x)
108    }
109
110    /// モノイドを重みとして載せている場合における、`[x_begin, x_end)`, `[y_begin, y_end)`内の点の重みの和を求める
111    pub fn rect_sum_monoid<R1: RangeBounds<T>, R2: RangeBounds<T>>(
112        &self,
113        x_range: R1,
114        y_range: R2,
115    ) -> M::Target {
116        let (xl, xr) = self.get_pos_range(x_range);
117        let (y_low, y_hi) = self.get_num_range(y_range);
118        self.wm.rect_sum_monoid(xl, xr, y_low, y_hi)
119    }
120
121    /// 群を重みとして載せている場合における、`[x_begin, x_end)`, `[y_begin, y_end)`内の点の重みの和を求める  
122    /// prefix_sumを二度求める非再帰の実装なのでモノイド版より定数倍が良いはず
123    pub fn rect_sum_group<R1: RangeBounds<T>, R2: RangeBounds<T>>(
124        &self,
125        x_range: R1,
126        y_range: R2,
127    ) -> M::Target
128    where
129        M: Group,
130    {
131        let (xl, xr) = self.get_pos_range(x_range);
132        let (y_low, y_hi) = self.get_num_range(y_range);
133        self.wm.rect_sum_group(xl, xr, y_low, y_hi)
134    }
135}

137/// Wavelet Matrix にビットごとのSegment Treeを追加したもの  
138struct WaveletMatrixSegTree<M: Monoid + Commutative> {
139    len: usize,
140    /// indices[i] = 下からiビット目に関する索引
141    indices: Vec<BitDict>,
142    /// ビットごとのSegTree
143    segtree_per_bit: Vec<SegTree<M>>,
144}
145
146impl<M: Monoid + Commutative> WaveletMatrixSegTree<M> {
147    /// `compressed_list[x] = y` が点(x, y)に、`weight_list[x] = w` が点(x, y)の重みwに対応する  
148    /// compressed_listには今後更新クエリのある(x, y)も含める  
149    /// compressed_listは座標圧縮されていることを期待する  
150    /// xは重複不可なので、順番を振りなおしてもらうことになる  
151    /// 全て0以上
152    pub fn from_weight(compressed_list: &[usize], weight_list: &[M::Target]) -> Self {
153        assert_eq!(compressed_list.len(), weight_list.len());
154        let len = compressed_list.len();
155        let upper_bound = *compressed_list.iter().max().unwrap_or(&0) + 1;
156        let log = ceil_log2(upper_bound as u32 + 1) as usize;
157        let mut indices = vec![BitDict::new(len); log];
158        // 注目する桁のbitが0となる数、1となる数
159        let mut tmp = vec![Vec::with_capacity(len); 2];
160        let mut list = compressed_list.to_vec();
161        let mut weight_list = weight_list.to_vec();
162        let mut tmp_weight = vec![Vec::with_capacity(len); 2];
163        let mut segtree_per_bit = Vec::with_capacity(log);
164        for (ln, index) in indices.iter_mut().enumerate().rev() {
165            for (x, (y, w)) in list.drain(..).zip(weight_list.drain(..)).enumerate() {
166                if (y >> ln) & 1 == 1 {
167                    index.set(x);
168                    tmp[1].push(y);
169                    tmp_weight[1].push(w);
170                } else {
171                    tmp[0].push(y);
172                    tmp_weight[0].push(w);
173                }
174            }
175            index.build();
176            list.append(&mut tmp[0]);
177            list.append(&mut tmp[1]);
178            weight_list.append(&mut tmp_weight[0]);
179            weight_list.append(&mut tmp_weight[1]);
180            segtree_per_bit.push(SegTree::from(&weight_list));
181        }
182        segtree_per_bit.reverse();
183        Self {
184            len,
185            indices,
186            segtree_per_bit,
187        }
188    }
189
190    /// x座標が[begin, end)内、y座標はupper未満の点の重みの和を求める
191    pub fn prefix_rect_sum(&self, mut begin: usize, mut end: usize, upper: usize) -> M::Target {
192        if upper == 0 {
193            return M::id_element();
194        }
195        let mut ret = M::id_element();
196        for (ln, index) in self.indices.iter().enumerate().rev() {
197            let bit = (upper >> ln) & 1;
198            let rank1_begin = index.rank_1(begin);
199            let rank1_end = index.rank_1(end);
200            let rank0_begin = begin - rank1_begin;
201            let rank0_end = end - rank1_end;
202            if bit == 1 {
203                ret = M::binary_operation(
204                    &ret,
205                    &self.segtree_per_bit[ln].prod(rank0_begin..rank0_end),
206                );
207                begin = index.rank0_all() + rank1_begin;
208                end = index.rank0_all() + rank1_end;
209            } else {
210                begin = rank0_begin;
211                end = rank0_end;
212            }
213        }
214        ret
215    }
216
217    /// 群を重みとして載せている場合における、`[x_begin, x_end)`, `[y_begin, y_end)`内の点の重みの和を求める  
218    /// prefix_sumを二度求めて引く 非再帰なので定数倍が良いはず
219    pub fn rect_sum_group(
220        &self,
221        x_begin: usize,
222        x_end: usize,
223        y_begin: usize,
224        y_end: usize,
225    ) -> M::Target
226    where
227        M: Group,
228    {
229        let s2 = self.prefix_rect_sum(x_begin, x_end, y_end);
230        let s1 = self.prefix_rect_sum(x_begin, x_end, y_begin);
231        M::binary_operation(&M::inverse(&s1), &s2)
232    }
233
234    /// モノイドを重みとして載せている場合における、`[x_begin, x_end)`, `[y_begin, y_end)`内の点の重みの和を求める  
235    /// 完全に覆うか外れるかするまで再帰的に二冪の長さの区間に分けていく
236    pub fn rect_sum_monoid(
237        &self,
238        x_begin: usize,
239        x_end: usize,
240        y_begin: usize,
241        y_end: usize,
242    ) -> M::Target {
243        let mut ret = M::id_element();
244        let ln = self.indices.len();
245        self.dfs(&mut ret, ln, x_begin, x_end, 0, 1 << ln, y_begin, y_end);
246        ret
247    }
248
249    #[allow(clippy::too_many_arguments)]
250    fn dfs(
251        &self,
252        ret: &mut M::Target,
253        ln: usize,
254        xl: usize,
255        xr: usize,
256        yl: usize,
257        yr: usize,
258        y_low: usize,
259        y_hi: usize,
260    ) {
261        assert_eq!(yr - yl, 1 << ln);
262        if y_hi <= yl || yr <= y_low {
263            return;
264        }
265        if y_low <= yl && yr <= y_hi {
266            *ret = M::binary_operation(ret, &self.segtree_per_bit[ln].prod(xl..xr));
267            return;
268        }
269        let ln = ln - 1;
270        let rank1_xl = self.indices[ln].rank_1(xl);
271        let rank1_xr = self.indices[ln].rank_1(xr);
272        let rank0_all = self.indices[ln].rank0_all();
273        let rank0_xl = xl - rank1_xl;
274        let rank0_xr = xr - rank1_xr;
275        let ymid = (yl + yr) / 2;
276        self.dfs(ret, ln, rank0_xl, rank0_xr, yl, ymid, y_low, y_hi);
277        self.dfs(
278            ret,
279            ln,
280            rank0_all + rank1_xl,
281            rank0_all + rank1_xr,
282            ymid,
283            yr,
284            y_low,
285            y_hi,
286        );
287    }
288
289    pub fn set(&mut self, mut x: usize, new_val: M::Target) {
290        assert!(x < self.len);
291        for (ln, index) in self.indices.iter().enumerate().rev() {
292            let bit = index.access(x);
293            if bit {
294                x = index.rank0_all() + index.rank_1(x);
295            } else {
296                x = index.rank_0(x);
297            }
298            self.segtree_per_bit[ln].set(x, new_val.clone());
299        }
300    }
301
302    pub fn get_weight(&self, x: usize) -> M::Target {
303        assert!(x < self.len);
304        let index = self.indices.last().unwrap();
305        let x = if index.access(x) {
306            index.rank0_all() + index.rank_1(x)
307        } else {
308            index.rank_0(x)
309        };
310        self.segtree_per_bit.last().unwrap().get(x)
311    }
312}

319    // 加算群
320    #[derive(Debug)]
321    struct AddGroup;
322    impl Monoid for AddGroup {
323        type Target = i64;
324        fn id_element() -> Self::Target {
325            0
326        }
327        fn binary_operation(a: &Self::Target, b: &Self::Target) -> Self::Target {
328            a + b
329        }
330    }
331    impl Commutative for AddGroup {}
332    impl Group for AddGroup {
333        fn inverse(a: &Self::Target) -> Self::Target {
334            -a
335        }
336    }

339    fn test_static_rect_sum() {
340        use wavelet_matrix_cum_sum::WaveletMatrixCumSum;
341        let mut rng = thread_rng();
342        const SIZE: usize = 100000;
343        let num_list: Vec<usize> = (0..SIZE).map(|_| rng.gen_range(0..SIZE)).collect();
344        let wm_cum_sum = WaveletMatrixCumSum::new(&num_list, &num_list);
345        let num_list_i64: Vec<i64> = num_list.iter().map(|i| *i as i64).collect();
346        let wm_seg = WaveletMatrixSegTree::<AddGroup>::from_weight(&num_list, &num_list_i64);
347
348        for _ in 0..SIZE {
349            let xl = rng.gen_range(0..SIZE);
350            let xr = rng.gen_range(xl..SIZE);
351            let yl = rng.gen_range(0..SIZE);
352            let yr = rng.gen_range(yl..SIZE);
353            let cum_sum_ans = wm_cum_sum.rect_sum(xl..xr, yl..yr) as i64;
354            assert_eq!(cum_sum_ans, wm_seg.rect_sum_group(xl, xr, yl, yr));
355            assert_eq!(cum_sum_ans, wm_seg.rect_sum_monoid(xl, xr, yl, yr))

359    #[test]
360    fn test_point_add_rect_sum() {
361        let mut rng = thread_rng();
362        const SIZE: usize = 10000;
363        let mut weight_list = (0..SIZE)
364            .map(|_| rng.gen_range(-1000_000_000..=1000_000_000))
365            .collect::<Vec<i64>>();
366        let y_list = (0..SIZE)
367            .map(|_| rng.gen_range(0..=SIZE))
368            .collect::<Vec<usize>>();
369        let mut wm = WaveletMatrixSegTree::<AddGroup>::from_weight(&y_list, &weight_list);
370        for _ in 0..1000 {
371            let x_left = rng.gen_range(0..=SIZE);
372            let x_right = rng.gen_range(x_left..=SIZE);
373            let y_left = rng.gen_range(0..=SIZE);
374            let y_right = rng.gen_range(y_left..=SIZE);
375            let real_sum = weight_list
376                .iter()
377                .enumerate()
378                .skip(x_left)
379                .take(x_right - x_left)
380                .filter(|&(i, _)| y_left <= y_list[i] && y_list[i] < y_right)
381                .map(|(_, &w)| w)
382                .sum::<i64>();
383            assert_eq!(
384                wm.rect_sum_group(x_left, x_right, y_left, y_right),
385                real_sum
386            );
387            assert_eq!(
388                wm.rect_sum_monoid(x_left, x_right, y_left, y_right),
389                real_sum
390            );
391            let pos = rng.gen_range(0..SIZE);
392            let new_weight = rng.gen_range(-1000_000_000..=1000_000_000);
393            weight_list[pos] = new_weight;
394            wm.set(pos, new_weight);
395        }
396    }
