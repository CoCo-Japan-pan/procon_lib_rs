1//! 部分永続UnionFind  
2//! 全バージョンにアクセス可能であり、最新バージョンのみ更新可能なUnionFind  
3//! 経路圧縮はできないのでクエリはO(logN)  
4//! <https://blog.tiramister.net/posts/persistent-unionfind/>
5
6pub struct PartiallyPersistentUnionFind {
7    size: usize,
8    /// 現在時刻
9    now: usize,
10    /// 各ノードの親(根の場合は自身)
11    par: Vec<usize>,
12    /// 各ノードの親の更新時刻
13    time: Vec<usize>,
14    /// (時刻、集合のサイズ)の記録
15    num: Vec<Vec<(usize, usize)>>,
16}
17
18impl PartiallyPersistentUnionFind {
19    pub fn new(size: usize) -> Self {
20        PartiallyPersistentUnionFind {
21            size,
22            now: 0,
23            par: (0..size).collect(),
24            time: vec![usize::MAX; size],
25            num: vec![vec![(0, 1)]; size],
26        }
27    }
28
29    /// 時刻tにおいてxの属する集合の根を返す
30    pub fn leader(&self, x: usize, t: usize) -> usize {
31        assert!(x < self.size);
32        assert!(t <= self.now);
33        if self.time[x] > t {
34            x
35        } else {
36            self.leader(self.par[x], t)
37        }
38    }
39
40    /// 時刻tにおいて、xとyが同じ集合に属するかどうかを返す
41    pub fn same(&self, x: usize, y: usize, t: usize) -> bool {
42        self.leader(x, t) == self.leader(y, t)
43    }
44
45    /// xとyが属する集合を併合し、時間を進める  
46    /// 最新時刻を返す
47    pub fn merge(&mut self, x: usize, y: usize) -> usize {
48        assert!(x < self.size);
49        assert!(y < self.size);
50        let (x, y) = (self.leader(x, self.now), self.leader(y, self.now));
51        self.now += 1;
52        if x == y {
53            return self.now;
54        }
55        let x_size = self.num[x].last().unwrap().1;
56        let y_size = self.num[y].last().unwrap().1;
57        let (x, y) = if x_size < y_size { (y, x) } else { (x, y) };
58
59        let new_size = x_size + y_size;
60        self.num[x].push((self.now, new_size));
61
62        // xにyをくっつける
63        self.par[y] = x;
64        self.time[y] = self.now;
65
66        self.now
67    }
68
69    /// 時刻tにおいてxの属する集合のサイズを返す
70    pub fn size(&self, x: usize, t: usize) -> usize {
71        assert!(x < self.size);
72        assert!(t <= self.now);
73        let x = self.leader(x, t);
74
75        let mut ok = 0;
76        let mut ng = self.num[x].len();
77        while ng - ok > 1 {
78            let mid = (ok + ng) / 2;
79            if self.num[x][mid].0 <= t {
80                ok = mid;
81            } else {
82                ng = mid;
83            }
84        }
85        self.num[x][ok].1
86    }
87}

89#[cfg(test)]
90mod tests {
91    use super::*;
92    use rand::prelude::*;
93    use unionfind::UnionFind;
94
95    #[test]
96    fn test() {
97        let mut rng = rand::thread_rng();
98        const SIZE: usize = 100;
99        let mut puf = PartiallyPersistentUnionFind::new(SIZE);
100        let mut ufs = vec![UnionFind::new(SIZE)];
101        for _ in 0..SIZE * 2 {
102            let a = rng.gen_range(0..SIZE);
103            let b = rng.gen_range(0..SIZE);
104            let mut last = ufs.last().unwrap().clone();
105            last.merge(a, b);
106            ufs.push(last);
107            puf.merge(a, b);
108        }
109        for t in 0..=SIZE * 2 {
110            for i in 0..SIZE {
111                for j in 0..SIZE {
112                    assert_eq!(ufs[t].same(i, j), puf.same(i, j, t));
113                }
114            }
115            for i in 0..SIZE {
116                assert_eq!(ufs[t].size(i), puf.size(i, t));
117            }
118        }
119    }
