1//! 蟻本p165をもとにしている  
2//! fenwick tree を二つ用いて、区間加算、区間和クエリを処理します  
3
4use fenwick_tree::FenwickTree;
5use std::ops::{Add, AddAssign, Bound::*, Mul, Neg, RangeBounds, Sub};
6
7pub struct RAQRSQ<
8    T: Clone
9        + Add<Output = T>
10        + AddAssign
11        + Sub<Output = T>
12        + Neg<Output = T>
13        + From<u32>
14        + Mul<Output = T>,
15> {
16    range_size: usize,
17    ft1: FenwickTree<T>,
18    ft2: FenwickTree<T>,
19}
20
21impl<
22        T: Clone
23            + Add<Output = T>
24            + AddAssign
25            + Sub<Output = T>
26            + Neg<Output = T>
27            + From<u32>
28            + Mul<Output = T>,
29    > RAQRSQ<T>
30{
31    pub fn new(size: usize, zero: T) -> Self {
32        Self {
33            range_size: size,
34            ft1: FenwickTree::new(size + 1, zero.clone()),
35            ft2: FenwickTree::new(size + 1, zero),
36        }
37    }
38
39    /// 1点加算
40    pub fn add_point(&mut self, idx: usize, val: T) {
41        assert!(idx < self.range_size);
42        self.ft1.add(idx, val);
43    }
44
45    fn get_range<R: RangeBounds<usize>>(&self, range: R) -> (usize, usize) {
46        let begin = match range.start_bound() {
47            Included(&s) => s,
48            Excluded(&s) => s + 1,
49            Unbounded => 0,
50        };
51        let end = match range.end_bound() {
52            Included(&e) => e + 1,
53            Excluded(&e) => e,
54            Unbounded => self.range_size,
55        };
56        assert!(begin <= end && end <= self.range_size);
57        (begin, end)
58    }
59
60    /// 区間加算
61    pub fn add<R: RangeBounds<usize>>(&mut self, range: R, val: T) {
62        let (begin, end) = self.get_range(range);
63        self.ft1.add(begin, -val.clone() * (begin as u32).into());
64        self.ft2.add(begin, val.clone());
65        self.ft1.add(end, val.clone() * (end as u32).into());
66        self.ft2.add(end, -val);
67    }
68
69    pub fn sum<R: RangeBounds<usize>>(&self, range: R) -> T {
70        let (begin, end) = self.get_range(range);
71        self.sum_from_first(end) - self.sum_from_first(begin)
72    }
73
74    fn sum_from_first(&self, idx: usize) -> T {
75        assert!(idx <= self.range_size);
76        self.ft1.sum(0..idx) + self.ft2.sum(0..idx) * (idx as u32).into()
77    }
78}
