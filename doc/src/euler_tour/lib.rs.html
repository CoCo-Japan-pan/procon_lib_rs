1//! 頂点に着目したオイラーツアー  
2//! LCAをRMQに帰着させて求められる  
3//! SparseTableを用いるので前時間`O(NlogN)`、クエリ時間`O(1)`  
4use algebra::{IdempotentMonoid, Monoid};
5use sparse_table::SparseTable;
6
7#[derive(Debug)]
8pub struct MinMonoid;
9impl Monoid for MinMonoid {
10    type Target = (usize, usize);
11    fn id_element() -> Self::Target {
12        (usize::MAX, usize::MAX)
13    }
14    fn binary_operation(a: &Self::Target, b: &Self::Target) -> Self::Target {
15        *a.min(b)
16    }
17}
18impl IdempotentMonoid for MinMonoid {}
19
20#[derive(Debug)]
21pub struct EulerTour {
22    /// 頂点に着目したオイラーツアー  
23    /// 各辺を2回ずつ通るので、サイズは`2|E| + 1 = "2|V| - 1`
24    pub euler_tour_vertex: Vec<usize>,
25    /// 各頂点の深さ
26    pub depth: Vec<usize>,
27    /// オイラーツアーにおいて、各頂点が最初に出現するインデックス
28    pub first_occurrence: Vec<usize>,
29    /// オイラーツアーにおいて、各頂点が最後に出現するインデックス
30    pub last_occurrence: Vec<usize>,
31    /// (深さ、頂点)の配列から構成されるSparseTable  
32    /// first_occurenceの[最小、最大]の範囲で区間積を取ることで、(lcaの深さ、lcaの頂点)を求められる  
33    sparse_table: SparseTable<MinMonoid>,
34}
35
36impl EulerTour {
37    /// SparseTableを構築しているので、`O(NlogN)`
38    pub fn new(graph: &[Vec<usize>], root: usize) -> Self {
39        let n = graph.len();
40        struct Cls<'a> {
41            graph: &'a [Vec<usize>],
42            euler_tour_vertex: Vec<usize>,
43            depth: Vec<usize>,
44        }
45        let mut cls = Cls {
46            graph,
47            euler_tour_vertex: Vec::with_capacity(2 * n - 1),
48            depth: vec![0; n],
49        };
50        fn dfs(cls: &mut Cls, v: usize, p: usize) {
51            cls.euler_tour_vertex.push(v);
52            for &nv in &cls.graph[v] {
53                if nv == p {
54                    continue;
55                }
56                cls.depth[nv] = cls.depth[v] + 1;
57                dfs(cls, nv, v);
58                cls.euler_tour_vertex.push(v);
59            }
60        }
61        dfs(&mut cls, root, n);
62        let mut first_occurrence = vec![usize::MAX; n];
63        let mut last_occurrence = vec![0; n];
64        for (i, &v) in cls.euler_tour_vertex.iter().enumerate() {
65            first_occurrence[v] = first_occurrence[v].min(i);
66            last_occurrence[v] = i;
67        }
68        // オイラーツアーの深さと頂点のペアからなる配列を作成
69        let depth_vertex = cls
70            .euler_tour_vertex
71            .iter()
72            .map(|&v| (cls.depth[v], v))
73            .collect();
74        let sparse_table = SparseTable::new(depth_vertex);
75        Self {
76            euler_tour_vertex: cls.euler_tour_vertex,
77            depth: cls.depth,
78            first_occurrence,
79            last_occurrence,
80            sparse_table,
81        }
82    }
83
84    /// SparseTableを用いているので、`O(1)`
85    pub fn lca(&self, u: usize, v: usize) -> usize {
86        let l = self.first_occurrence[u];
87        let r = self.first_occurrence[v];
88        let (l, r) = (l.min(r), l.max(r));
89        self.sparse_table.prod(l..=r).1
90    }
91
92    pub fn lca_multiple(&self, vertex_list: &[usize]) -> usize {
93        let l = vertex_list
94            .iter()
95            .map(|&v| self.first_occurrence[v])
96            .min()
97            .unwrap();
98        let r = vertex_list
99            .iter()
100            .map(|&v| self.first_occurrence[v])
101            .max()
102            .unwrap();
103        self.sparse_table.prod(l..=r).1
104    }
105}
