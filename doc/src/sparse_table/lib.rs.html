1//! 冪等モノイドが乗った静的な区間クエリを処理する  
2//! Disjoint Sparse Table に比べて定数倍早い  
3
4use algebra::IdempotentMonoid;
5use monoid_utils::{MaxMonoid, MinMonoid};
6use std::ops::RangeBounds;
7
8pub type MinSparseTable<T> = SparseTable<MinMonoid<T>>;
9pub type MaxSparseTable<T> = SparseTable<MaxMonoid<T>>;
10
11#[derive(Debug, Clone)]
12pub struct SparseTable<M: IdempotentMonoid> {
13    range_size: usize,
14    data: Vec<Vec<M::Target>>,
15}
16
17impl<M: IdempotentMonoid> SparseTable<M> {
18    /// `O(nlogn)`
19    pub fn new(v: Vec<M::Target>) -> Self {
20        let range_size = v.len();
21        let mut data = vec![v];
22        let log_floor = if range_size == 0 {
23            0
24        } else {
25            range_size.ilog2() as usize
26        };
27        for i in 1..=log_floor {
28            let mut row = vec![M::id_element(); range_size - (1 << i) + 1];
29            for (j, r) in row.iter_mut().enumerate() {
30                *r = M::binary_operation(&data[i - 1][j], &data[i - 1][j + (1 << (i - 1))]);
31            }
32            data.push(row);
33        }
34        Self { range_size, data }
35    }
36
37    /// `O(1)`
38    pub fn prod<R: RangeBounds<usize>>(&self, range: R) -> M::Target {
39        use std::ops::Bound::*;
40        let l = match range.start_bound() {
41            Included(&l) => l,
42            Excluded(&l) => l + 1,
43            Unbounded => 0,
44        };
45        let r = match range.end_bound() {
46            Included(&r) => r + 1,
47            Excluded(&r) => r,
48            Unbounded => self.range_size,
49        };
50        assert!(l <= r && r <= self.range_size);
51        if l == r {
52            return M::id_element();
53        }
54        let k = (r - l).ilog2() as usize;
55        M::binary_operation(&self.data[k][l], &self.data[k][r - (1 << k)])
56    }
57}
