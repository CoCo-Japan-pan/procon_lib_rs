1//! [abc345e](https://atcoder.jp/contests/abc345/tasks/abc345_e) でTop2のみ保持するものが欲しくなったのでライブラリ化
2
3#[derive(Debug, Clone, Copy)]
4enum Inner<K: Copy, V: Copy> {
5    Empty,
6    One(K, V),
7    Two([(K, V); 2]),
8}
9
10/// Top2(大きい順)までのMapを持つ ただし同じKeyは一つまで
11#[derive(Debug, Clone, Copy)]
12pub struct Top2Map<K: Eq + Copy, V: Ord + Copy> {
13    map: Inner<K, V>,
14}
15
16impl<K: Eq + Copy, V: Ord + Copy> Default for Top2Map<K, V> {
17    fn default() -> Self {
18        Self::new()
19    }
20}
21
22impl<K: Eq + Copy, V: Ord + Copy> Top2Map<K, V> {
23    pub fn new() -> Self {
24        Self { map: Inner::Empty }
25    }
26    #[allow(clippy::collapsible_else_if)]
27    pub fn insert(&mut self, key: K, value: V) {
28        match self.map {
29            Inner::Empty => {
30                self.map = Inner::One(key, value);
31            }
32            Inner::One(k, v) => {
33                if key == k {
34                    self.map = Inner::One(key, v.max(value));
35                } else {
36                    if value > v {
37                        self.map = Inner::Two([(key, value), (k, v)]);
38                    } else {
39                        self.map = Inner::Two([(k, v), (key, value)]);
40                    }
41                }
42            }
43            Inner::Two([(k1, v1), (k2, v2)]) => {
44                if key == k1 {
45                    self.map = Inner::Two([(key, v1.max(value)), (k2, v2)]);
46                } else if key == k2 {
47                    let new_k2_val = v2.max(value);
48                    if new_k2_val > v1 {
49                        self.map = Inner::Two([(k2, new_k2_val), (k1, v1)]);
50                    } else {
51                        self.map = Inner::Two([(k1, v1), (k2, new_k2_val)]);
52                    }
53                } else {
54                    if value > v1 {
55                        self.map = Inner::Two([(key, value), (k1, v1)]);
56                    } else if value > v2 {
57                        self.map = Inner::Two([(k1, v1), (key, value)]);
58                    }
59                }
60            }
61        }
62    }
63    pub fn first(&self) -> Option<(K, V)> {
64        match self.map {
65            Inner::Empty => None,
66            Inner::One(k, v) => Some((k, v)),
67            Inner::Two([(k, v), _]) => Some((k, v)),
68        }
69    }
70    pub fn second(&self) -> Option<(K, V)> {
71        match self.map {
72            Inner::Empty => None,
73            Inner::One(_, _) => None,
74            Inner::Two([_, (k, v)]) => Some((k, v)),
75        }
76    }
77}
