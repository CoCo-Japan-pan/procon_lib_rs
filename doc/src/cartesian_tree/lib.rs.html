1//! [Cartesian-tree](https://drken1215.hatenablog.com/entry/2023/10/19/025800)  
2//! 数列のmin/maxで左右に分割していく分割統治法と相性がよい  
3
4#[derive(Debug, Clone)]
5pub struct CartesianTree {
6    pub root: usize,
7    pub parent: Vec<Option<usize>>,
8    pub left_child: Vec<Option<usize>>,
9    pub right_child: Vec<Option<usize>>,
10}
11
12impl CartesianTree {
13    /// min_rootがtrueなら最小値を根に、falseなら最大値を根にする  
14    /// tie-breakについては任意に順序をつけて二分木を保つ実装にしているので注意  
15    pub fn new<T: PartialOrd>(list: &[T], min_root: bool) -> Self {
16        let n = list.len();
17        let mut parent = vec![None; n];
18        let mut stack = Vec::with_capacity(n);
19        let cmp = if min_root { |a, b| a > b } else { |a, b| a < b };
20        for (i, a) in list.iter().enumerate() {
21            let mut prev = None;
22            while let Some(&j) = stack.last() {
23                if cmp(&list[j], a) {
24                    prev = stack.pop();
25                } else {
26                    break;
27                }
28            }
29            if let Some(prev) = prev {
30                parent[prev] = Some(i);
31            }
32            if let Some(&last) = stack.last() {
33                parent[i] = Some(last);
34            }
35            stack.push(i);
36        }
37        let root = stack[0];
38        let mut left_child = vec![None; n];
39        let mut right_child = vec![None; n];
40        for (i, p) in parent.iter().enumerate() {
41            if let Some(p) = *p {
42                if i < p {
43                    left_child[p] = Some(i);
44                } else {
45                    right_child[p] = Some(i);
46                }
47            }
48        }
49        Self {
50            root,
51            parent,
52            left_child,
53            right_child,
54        }
55    }
56}
