1//! b-flowに帰着することで負辺削除や、最小流量制限を機械的に扱えるようにする  
2//! 内部でs-t間の最小費用流を求めるために、atcoderのライブラリを利用している
3
4use atcoder_mincostflow::MinCostFlowGraph;
5use internal_type_traits::Integral;
6use std::ops::Neg;
7
8#[derive(Debug, Clone, Default)]
9pub struct MinCostBFlowResult<T> {
10    /// もしs-tを指定している場合、その間に流れる流量 (ただのb-flowの場合は0)
11    pub st_flow: T,
12    /// 総コスト
13    pub cost: T,
14    /// 復元した流量
15    pub flow: Vec<T>,
16    /// ポテンシャル
17    pub potential: Vec<T>,
18}
19
20#[derive(Debug, Clone)]
21pub struct MinCostBFlow<T: Integral + Neg<Output = T>> {
22    result: MinCostBFlowResult<T>,
23    size: usize,
24    mcf: MinCostFlowGraph<T>,
25    b_list: Vec<T>,
26    /// 負辺の場合は逆にしている 負辺の各idごとに、逆にしていればtrue
27    rev: Vec<bool>,
28}
29
30impl<T: Integral + Neg<Output = T>> MinCostBFlow<T> {
31    pub fn new(n: usize) -> Self {
32        Self {
33            result: MinCostBFlowResult::default(),
34            size: n,
35            mcf: MinCostFlowGraph::new(n + 2),
36            b_list: vec![T::zero(); n],
37            rev: vec![],
38        }
39    }
40
41    /// `from -> to` に、`lower <= cap <= upper` の流量制限がある辺を張る  
42    /// 負辺の場合はあらかじめupperだけ流すので、ここはINFにせず、オーバフローしない上限を指定する！  
43    /// 流量が増えるので、TLEする場合、負辺はうまく下駄をはかせる等の対処が必要かも
44    pub fn add_edge(
45        &mut self,
46        mut from: usize,
47        mut to: usize,
48        mut lower: T,
49        mut upper: T,
50        mut cost: T,
51    ) -> usize {
52        assert!(from < self.size);
53        assert!(to < self.size);
54        assert!(T::zero() <= lower);
55        assert!(lower <= upper);
56        let minus_edge = cost < T::zero();
57        self.rev.push(minus_edge);
58        // 負辺の場合は最大まであらかじめ流し、逆の辺を張る
59        if minus_edge {
60            std::mem::swap(&mut from, &mut to);
61            (lower, upper) = (-upper, -lower);
62            cost = -cost;
63        }
64        // from -> to にあらかじめlowerだけ流しておく
65        self.b_list[from] -= lower;
66        self.b_list[to] += lower;
67        self.result.flow.push(lower);
68        self.result.cost += lower * cost;
69        self.mcf.add_edge(from, to, upper - lower, cost)
70    }
71
72    /// 頂点vにsupply分の湧き出しを追加
73    pub fn add_supply(&mut self, v: usize, supply: T) {
74        assert!(v < self.size);
75        assert!(supply >= T::zero());
76        self.b_list[v] += supply;
77    }
78
79    /// 頂点vにdemand分の吸い込みを追加
80    pub fn add_demand(&mut self, v: usize, demand: T) {
81        assert!(v < self.size);
82        assert!(demand >= T::zero());
83        self.b_list[v] -= demand;
84    }
85
86    /// 超頂点を用意してst-flowに帰着し流し、総コストも更新  
87    /// bの正の和と負の絶対値の和が等しくない場合はfalseを返す  
88    /// このとき最大まで流せればtrueを返す
89    fn reduce_to_st_flow(&mut self) -> bool {
90        let dummy_source = self.size;
91        let dummy_sink = self.size + 1;
92        let mut positive_sum = T::zero();
93        let mut negative_sum = T::zero();
94        for (v, &b) in self.b_list.iter().enumerate() {
95            use std::cmp::Ordering::*;
96            match b.cmp(&T::zero()) {
97                Less => {
98                    self.mcf.add_edge(v, dummy_sink, -b, T::zero());
99                    negative_sum += -b;
100                }
101                Greater => {
102                    self.mcf.add_edge(dummy_source, v, b, T::zero());
103                    positive_sum += b;
104                }
105                Equal => {}
106            }
107        }
108        if positive_sum != negative_sum {
109            return false;
110        }
111        let (flow, cost) = self.mcf.flow(dummy_source, dummy_sink, positive_sum);
112        self.result.cost += cost;
113        flow == positive_sum
114    }
115
116    fn recover_flow_potential(&mut self) {
117        let edges = self.mcf.edges();
118        let mut potential_edges = vec![];
119        for ((res_flow, edge), rev) in self.result.flow.iter_mut().zip(edges).zip(&self.rev) {
120            *res_flow += edge.flow;
121            if *rev {
122                *res_flow = -*res_flow;
123            }
124
125            // 相補性条件
126            if edge.flow < edge.cap {
127                potential_edges.push((edge.from, edge.to, edge.cost));
128            }
129            if edge.flow > T::zero() {
130                potential_edges.push((edge.to, edge.from, -edge.cost));
131            }
132        }
133
134        // ベルマンフォード
135        self.result.potential.resize(self.size, T::zero());
136        for _ in 0..self.size {
137            for &(from, to, cost) in &potential_edges {
138                let old = self.result.potential[to];
139                let new = self.result.potential[from] + cost;
140                self.result.potential[to] = old.min(new);
141            }
142        }
143    }
144
145    /// 最小費用b-flowを解く  
146    /// infeasible(実行不可能)ならNoneを返す  
147    /// 解ける場合はそのときの最小費用、復元した各辺の流量、各頂点のポテンシャルを返す
148    pub fn mincost_bflow(&mut self) -> Option<&MinCostBFlowResult<T>> {
149        if !self.reduce_to_st_flow() {
150            return None;
151        }
152        self.recover_flow_potential();
153        Some(&self.result)
154    }
155
156    /// sからtに自由なだけ流せる場合の最小費用b-flowを解く  
157    /// これはただtからsに無限容量、コスト0の辺を張ってb-flowを解くだけ  
158    /// bの条件を満たせない(infeasible)ならNoneを返す  
159    pub fn st_mincost_freeflow(&mut self, s: usize, t: usize) -> Option<&MinCostBFlowResult<T>> {
160        assert!(s < self.size && t < self.size && s != t);
161        let t_to_s_id = self.mcf.add_edge(t, s, T::max_value(), T::zero());
162        if !self.reduce_to_st_flow() {
163            return None;
164        }
165        self.result.st_flow = self.mcf.get_edge(t_to_s_id).flow;
166        self.recover_flow_potential();
167        Some(&self.result)
168    }
169
170    /// sからtに流せるだけ流さないといけない場合の最小費用b-flowを解く  
171    /// bの条件を満たせない(infeasible)ならNoneを返す  
172    pub fn st_mincost_maxflow(&mut self, s: usize, t: usize) -> Option<&MinCostBFlowResult<T>> {
173        assert!(s < self.size && t < self.size && s != t);
174        // まずsからtに自由に流せるときを求める
175        let t_to_s_id = self.mcf.add_edge(t, s, T::max_value(), T::zero());
176        if !self.reduce_to_st_flow() {
177            return None;
178        }
179        let first_flow = self.mcf.get_edge(t_to_s_id).flow;
180        // sに+無限、tに-無限と考えて流す
181        let (add_flow, add_cost) = self.mcf.flow(s, t, T::max_value());
182        self.result.st_flow = first_flow + add_flow;
183        self.result.cost += add_cost;
184        self.recover_flow_potential();
185        Some(&self.result)
186    }
