1//! Wavelet Matrix に Fenwick Tree (Binary Indexed Tree) を載せて、
2//! 1点更新、矩形和クエリを処理する  
3//! BITはSegment Treeより定数倍が軽いので、通常の意味での和を求める場合はこちらを推奨
4
5use bitdict::BitDict;
6use fenwick_tree::FenwickTree;
7use internal_bits::ceil_log2;
8use internal_type_traits::Integral;
9use std::ops::RangeBounds;
10
11/// 座標圧縮とx座標の重複除去を行うWrapper `T`が座標圧縮する型 `U`が重みの型
12pub struct WMFenwickWrapper<T: Integral, U: Integral> {
13    wm: WaveletMatrixFenwick<U>,
14    sorted_y: Vec<T>,
15    x_y: Vec<(T, T)>,
16}
17
18impl<T: Integral, U: Integral> WMFenwickWrapper<T, U> {
19    /// すべて単位元で初期化する場合
20    pub fn new(update_points: Vec<(T, T)>) -> Self {
21        Self::from_weight(update_points, &[])
22    }
23
24    /// update_pointsは更新クエリのある点の座標のリスト ただしinit_weightsの点も含める  
25    /// init_weightsは初期状態の点の座標と重みのリスト (x, y, w)  
26    /// もしinit_weightsの点が重複する場合は、それらmonoidの積として初期化するので注意(上書きしたい場合は事前に重複を消す前処理をしてください)
27    pub fn from_weight(mut update_points: Vec<(T, T)>, init_weights: &[(T, T, U)]) -> Self {
28        update_points.sort_unstable();
29        update_points.dedup();
30        let mut sorted_y = update_points
31            .iter()
32            .map(|(_, y)| y)
33            .copied()
34            .collect::<Vec<_>>();
35        sorted_y.sort_unstable();
36        let compressed_list = update_points
37            .iter()
38            .map(|(_, y)| sorted_y.binary_search(y).unwrap())
39            .collect::<Vec<_>>();
40        let mut weight_list = vec![U::zero(); update_points.len()];
41        for (x, y, w) in init_weights {
42            let idx = update_points
43                .binary_search(&(*x, *y))
44                .expect("init_weight points are not in update_points!!!");
45            weight_list[idx] += *w;
46        }
47        let wm = WaveletMatrixFenwick::<U>::from_weight(&compressed_list, &weight_list);
48        Self {
49            wm,
50            sorted_y,
51            x_y: update_points,
52        }
53    }
54
55    fn get_pos_range<R: RangeBounds<T>>(&self, range: R) -> (usize, usize) {
56        use std::ops::Bound::*;
57        let l = match range.start_bound() {
58            Included(&l) => l,
59            Excluded(&l) => l + T::one(),
60            Unbounded => T::min_value(),
61        };
62        let r = match range.end_bound() {
63            Included(&r) => r + T::one(),
64            Excluded(&r) => r,
65            Unbounded => T::max_value(),
66        };
67        assert!(l <= r);
68        let l = self.x_y.partition_point(|&(x, _)| x < l);
69        let r = self.x_y.partition_point(|&(x, _)| x < r);
70        (l, r)
71    }
72
73    fn get_num_range<R: RangeBounds<T>>(&self, range: R) -> (usize, usize) {
74        use std::ops::Bound::*;
75        let l = match range.start_bound() {
76            Included(&l) => l,
77            Excluded(&l) => l + T::one(),
78            Unbounded => T::min_value(),
79        };
80        let r = match range.end_bound() {
81            Included(&r) => r + T::one(),
82            Excluded(&r) => r,
83            Unbounded => T::max_value(),
84        };
85        assert!(l <= r);
86        let l = self.sorted_y.partition_point(|&y| y < l);
87        let r = self.sorted_y.partition_point(|&y| y < r);
88        (l, r)
89    }
90
91    /// 点(x, y)の重みをnew_valに更新する
92    pub fn set(&mut self, x: T, y: T, new_val: U) {
93        let x = self
94            .x_y
95            .binary_search(&(x, y))
96            .expect("(x, y) is not in update_queries!!!");
97        self.wm.set(x, new_val);
98    }
99
100    /// 点(x, y)の重みにadd_valを加算する
101    pub fn add(&mut self, x: T, y: T, add_val: U) {
102        let x = self
103            .x_y
104            .binary_search(&(x, y))
105            .expect("(x, y) is not in update_queries!!!");
106        self.wm.add(x, add_val);
107    }
108
109    /// 点(x, y)の重みを取得する
110    pub fn get(&self, x: T, y: T) -> U {
111        let Ok(x) = self.x_y.binary_search(&(x, y)) else {
112            return U::zero();
113        };
114        self.wm.get_weight(x)
115    }
116
117    /// 矩形区間内の点の重みの和を求める
118    pub fn rect_sum<R1: RangeBounds<T>, R2: RangeBounds<T>>(&self, x_range: R1, y_range: R2) -> U {
119        let (xl, xr) = self.get_pos_range(x_range);
120        let (y_low, y_hi) = self.get_num_range(y_range);
121        self.wm.rect_sum(xl, xr, y_low, y_hi)
122    }
123}

125struct WaveletMatrixFenwick<T: Integral> {
126    len: usize,
127    /// indices[i] = 下からiビット目に関する索引
128    indices: Vec<BitDict>,
129    /// ビットごとのFenwickTree
130    fenwick_per_bit: Vec<FenwickTree<T>>,
131}
132
133impl<T: Integral> WaveletMatrixFenwick<T> {
134    /// `compressed_list[x] = y` が点(x, y)に、`weight_list[x] = w` が点(x, y)の重みwに対応する  
135    /// compressed_listには今後更新クエリのある(x, y)も含める  
136    /// compressed_listは座標圧縮されていることを期待する  
137    /// xは重複不可なので、順番を振りなおしてもらうことになる  
138    /// 全て0以上
139    fn from_weight(compressed_list: &[usize], weight_list: &[T]) -> Self {
140        assert_eq!(compressed_list.len(), weight_list.len());
141        let len = compressed_list.len();
142        let upper_bound = *compressed_list.iter().max().unwrap_or(&0) + 1;
143        let log = ceil_log2(upper_bound as u32 + 1) as usize;
144        let mut indices = vec![BitDict::new(len); log];
145        // 注目する桁のbitが0となる数、1となる数
146        let mut tmp = vec![Vec::with_capacity(len); 2];
147        let mut list = compressed_list.to_vec();
148        let mut weight_list = weight_list.to_vec();
149        let mut tmp_weight = vec![Vec::with_capacity(len); 2];
150        let mut fenwick_per_bit = Vec::with_capacity(log);
151        for (ln, index) in indices.iter_mut().enumerate().rev() {
152            for (x, (y, w)) in list.drain(..).zip(weight_list.drain(..)).enumerate() {
153                if (y >> ln) & 1 == 1 {
154                    index.set(x);
155                    tmp[1].push(y);
156                    tmp_weight[1].push(w);
157                } else {
158                    tmp[0].push(y);
159                    tmp_weight[0].push(w);
160                }
161            }
162            index.build();
163            list.append(&mut tmp[0]);
164            list.append(&mut tmp[1]);
165            weight_list.append(&mut tmp_weight[0]);
166            weight_list.append(&mut tmp_weight[1]);
167            let mut cur_fenwick = FenwickTree::new(len, T::zero());
168            for (i, &w) in weight_list.iter().enumerate() {
169                cur_fenwick.add(i, w);
170            }
171            fenwick_per_bit.push(cur_fenwick);
172        }
173        fenwick_per_bit.reverse();
174        Self {
175            len,
176            indices,
177            fenwick_per_bit,
178        }
179    }
180
181    /// x座標が[begin, end)内、y座標はupper未満の点の重みの和を求める
182    fn prefix_rect_sum(&self, mut begin: usize, mut end: usize, upper: usize) -> T {
183        if upper == 0 {
184            return T::zero();
185        }
186        let mut ret = T::zero();
187        for (ln, index) in self.indices.iter().enumerate().rev() {
188            let bit = (upper >> ln) & 1;
189            let rank1_begin = index.rank_1(begin);
190            let rank1_end = index.rank_1(end);
191            let rank0_begin = begin - rank1_begin;
192            let rank0_end = end - rank1_end;
193            if bit == 1 {
194                ret += self.fenwick_per_bit[ln].sum(rank0_begin..rank0_end);
195                begin = index.rank0_all() + rank1_begin;
196                end = index.rank0_all() + rank1_end;
197            } else {
198                begin = rank0_begin;
199                end = rank0_end;
200            }
201        }
202        ret
203    }
204
205    fn rect_sum(&self, x_begin: usize, x_end: usize, y_begin: usize, y_end: usize) -> T {
206        self.prefix_rect_sum(x_begin, x_end, y_end) - self.prefix_rect_sum(x_begin, x_end, y_begin)
207    }
208
209    fn add(&mut self, mut x: usize, add_val: T) {
210        assert!(x < self.len);
211        for (ln, index) in self.indices.iter().enumerate().rev() {
212            let bit = index.access(x);
213            if bit {
214                x = index.rank0_all() + index.rank_1(x);
215            } else {
216                x = index.rank_0(x);
217            }
218            self.fenwick_per_bit[ln].add(x, add_val);
219        }
220    }
221
222    fn set(&mut self, mut x: usize, new_val: T) {
223        assert!(x < self.len);
224        for (ln, index) in self.indices.iter().enumerate().rev() {
225            let bit = index.access(x);
226            if bit {
227                x = index.rank0_all() + index.rank_1(x);
228            } else {
229                x = index.rank_0(x);
230            }
231            self.fenwick_per_bit[ln].set(x, new_val);
232        }
233    }
234
235    fn get_weight(&self, x: usize) -> T {
236        assert!(x < self.len);
237        let index = self.indices.last().unwrap();
238        let x = if index.access(x) {
239            index.rank0_all() + index.rank_1(x)
240        } else {
241            index.rank_0(x)
242        };
243        self.fenwick_per_bit.last().unwrap().get(x)
244    }
