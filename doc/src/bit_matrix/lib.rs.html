6pub trait BitMatrixOps {
7    fn add(lhs: &BitMatrix, rhs: &BitMatrix) -> BitMatrix;
8    fn mul(lhs: &BitMatrix, rhs: &BitMatrix) -> BitMatrix;
9}

11#[derive(Debug, Clone, PartialEq, Eq)]
12/// `+ = or, * = and`
13pub struct PlusOrMulAnd {}
14impl BitMatrixOps for PlusOrMulAnd {
15    fn add(lhs: &BitMatrix, rhs: &BitMatrix) -> BitMatrix {
16        assert_eq!(lhs.width, rhs.width);
17        assert_eq!(lhs.height, rhs.height);
18        let mut ret = BitMatrix::new(lhs.height, lhs.width);
19        for row in 0..lhs.height {
20            ret.data[row] = &lhs.data[row] | &rhs.data[row];
21        }
22        ret
23    }
24    fn mul(lhs: &BitMatrix, rhs: &BitMatrix) -> BitMatrix {
25        assert_eq!(lhs.width, rhs.height);
26        let mut ret = BitMatrix::new(lhs.height, rhs.width);
27        let rhs = rhs.transpose();
28        for i in 0..lhs.height {
29            for j in 0..rhs.height {
30                let val = lhs.data[i]
31                    .buffer()
32                    .iter()
33                    .zip(rhs.data[j].buffer())
34                    .fold(false, |acc, (l, r)| acc | ((l & r).count_ones() > 0));
35                ret.set(i, j, val);
36            }
37        }
38        ret
39    }
40}

42#[derive(Debug, Clone, PartialEq, Eq)]
43/// `+ = xor, * = and`
44pub struct PlusXorMulAnd {}
45impl BitMatrixOps for PlusXorMulAnd {
46    fn add(lhs: &BitMatrix, rhs: &BitMatrix) -> BitMatrix {
47        assert_eq!(lhs.width, rhs.width);
48        assert_eq!(lhs.height, rhs.height);
49        let mut ret = BitMatrix::new(lhs.height, lhs.width);
50        for row in 0..lhs.height {
51            ret.data[row] = &lhs.data[row] ^ &rhs.data[row];
52        }
53        ret
54    }
55    fn mul(lhs: &BitMatrix, rhs: &BitMatrix) -> BitMatrix {
56        assert_eq!(lhs.width, rhs.height);
57        let mut ret = BitMatrix::new(lhs.height, rhs.width);
58        let rhs = rhs.transpose();
59        for i in 0..lhs.height {
60            for j in 0..rhs.height {
61                let val = lhs.data[i]
62                    .buffer()
63                    .iter()
64                    .zip(rhs.data[j].buffer())
65                    .fold(false, |acc, (l, r)| acc ^ ((l & r).count_ones() & 1 > 0));
66                ret.set(i, j, val);
67            }
68        }
69        ret
70    }
71}

130impl BitMatrix {
131    pub fn new(height: usize, width: usize) -> Self {
132        Self {
133            height,
134            width,
135            data: vec![BitSet::new(width); height],
136        }
137    }
138
139    /// indexでアクセスしてもよい
140    pub fn get(&self, row: usize, col: usize) -> bool {
141        assert!(row < self.height && col < self.width);
142        self.data[row][col]
143    }
144
145    pub fn set(&mut self, row: usize, col: usize, b: bool) {
146        assert!(row < self.height && col < self.width);
147        self.data[row].set(col, b);
148    }
149
150    /// 掃き出し法(行基本変形)を行い、rankと線形独立な行のindexの集合を返す  
151    /// is_extendedがtrueの場合は拡大係数行列として扱い、係数行列の部分のrankを返す
152    pub fn gauss_jordan(&mut self, is_extended: bool) -> (usize, Vec<usize>) {
153        let mut rank = 0;
154        let col_end = if is_extended {
155            self.width - 1
156        } else {
157            self.width
158        };
159        let mut independent = vec![];
160        let mut ids = (0..self.height).collect::<Vec<_>>();
161        for col in 0..col_end {
162            let mut pivot = None;
163            for row in rank..self.height {
164                if self.data[row][col] {
165                    pivot = Some(row);
166                    break;
167                }
168            }
169            if let Some(pivot) = pivot {
170                self.data.swap(rank, pivot);
171                ids.swap(rank, pivot);
172                for row in 0..self.height {
173                    if row != rank && self.data[row][col] {
174                        unsafe {
175                            *self.data.as_mut_ptr().add(row) ^= &self.data[rank];
176                        }
177                    }
178                }
179                independent.push(ids[rank]);
180                rank += 1;
181            }
182        }
183        assert_eq!(rank, independent.len());
184        (rank, independent)
185    }
186
187    /// 連立一次方程式 Ax = bを解く(Aがselfの行列、bが引数のベクトル)  
188    /// 解が存在する場合は`Some((freedom, solution))`を返し、存在しない場合は`None`を返す  
189    /// freedomは解の自由度、solutionは解の一つを表すベクトル  
190    /// 解の個数は2^freedom個となる
191    pub fn linear_equation(&self, b: &[bool]) -> Option<(usize, Vec<bool>)> {
192        assert_eq!(self.height, b.len());
193        let mut mat = BitMatrix::new(self.height, self.width + 1);
194        #[allow(clippy::needless_range_loop)]
195        for i in 0..self.height {
196            for j in 0..self.width {
197                mat.set(i, j, self.get(i, j));
198            }
199            mat.set(i, self.width, b[i]);
200        }
201        let (rank, _) = mat.gauss_jordan(true);
202        for i in rank..self.height {
203            if mat.get(i, self.width) {
204                return None;
205            }
206        }
207        let mut ans = vec![false; self.width];
208        let mut cur_col = 0;
209        for r in 0..rank {
210            while cur_col < self.width && !mat.get(r, cur_col) {
211                cur_col += 1;
212            }
213            assert!(cur_col < self.width);
214            ans[cur_col] = mat.get(r, self.width);
215            cur_col += 1;
216        }
217        // 解の自由度
218        let freedom = self.width - rank;
219        Some((freedom, ans))
220    }
221
222    pub fn unit(n: usize) -> Self {
223        let mut res = Self::new(n, n);
224        for i in 0..n {
225            res.set(i, i, true);
226        }
227        res
228    }
229
230    pub fn transpose(&self) -> Self {
231        let mut res = Self::new(self.width, self.height);
232        for i in 0..self.height {
233            for j in 0..self.width {
234                res.set(j, i, self.get(i, j));
235            }
236        }
237        res
238    }
239
240    pub fn add<F: BitMatrixOps>(&self, rhs: &Self) -> Self {
241        F::add(self, rhs)
242    }
243
244    pub fn mul<F: BitMatrixOps>(&self, rhs: &Self) -> Self {
245        F::mul(self, rhs)
246    }
247
248    /// 行列のべき乗を計算する  
249    pub fn pow<F: BitMatrixOps>(&self, mut n: u64) -> Self {
250        assert_eq!(self.height, self.width);
251        let mut res = Self::unit(self.height);
252        let mut a = self.clone();
253        while n > 0 {
254            if (n & 1) == 1 {
255                res = F::mul(&res, &a);
256            }
257            a = F::mul(&a, &a);
258            n >>= 1;
259        }
260        res
261    }

276    #[test]
277    fn independent_test() {
278        let mut rng = thread_rng();
279        for _ in 0..10 {
280            let w = rng.gen_range(1..=10);
281            let h = rng.gen_range(w..=3 * w);
282            let mut mat = BitMatrix::new(h, w);
283            for i in 0..h {
284                for j in 0..w {
285                    mat.set(i, j, rng.gen());
286                }
287            }
288            let nums = {
289                let mut nums = vec![0; h];
290                for i in 0..h {
291                    for j in 0..w {
292                        if mat.get(i, j) {
293                            nums[i] |= 1 << j;
294                        }
295                    }
296                }
297                nums
298            };
299            let (rank, independent) = mat.gauss_jordan(false);
300            for i in 0..rank {
301                let cur_num = nums[independent[i]];
302                for bit in 0..(1 << rank) {
303                    let mut num = 0;
304                    for j in 0..rank {
305                        if j == i {
306                            continue;
307                        }
308                        if bit & (1 << j) > 0 {
309                            num ^= nums[independent[j]];
310                        }
311                    }
312                    assert_ne!(num, cur_num);
313                }
314            }
315        }
316    }

349    fn linear_eq_test() {
350        let mut rng = thread_rng();
351        let mut no_ans_cnt = 0;
352        for _ in 0..10 {
353            let n = rng.gen_range(1..=300);
354            let m = rng.gen_range(n..=300);
355            let mut mat = BitMatrix::new(n, m);
356            let mut b = vec![false; n];
357            for i in 0..n {
358                for j in 0..m {
359                    mat.set(i, j, rng.gen());
360                }
361                b[i] = rng.gen();
362            }
363            let Some((rank, ans)) = mat.linear_equation(&b) else {
364                no_ans_cnt += 1;
365                continue;
366            };
367            assert!(rank <= ans.len());
368            for i in 0..n {
369                let mut sum = false;
370                for j in 0..m {
371                    sum ^= mat.get(i, j) && ans[j];

376            // 行列の掛け算でも確認

378            let ans_mat = BitMatrix::from(vec![ans]).transpose();
