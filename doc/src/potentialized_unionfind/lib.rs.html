1//! ポテンシャル付きUnion-Find  
2//! 群を載せる  
3use algebra::Group;
4use std::cell::RefCell;
5use DiffOrSize::*;
6
7#[derive(Debug, Clone, Copy)]
8enum DiffOrSize<M> {
9    /// 親のノード番号と、親から見た差分
10    Diff(usize, M),
11    /// 自身が親なら、その集合のサイズを持つ
12    Size(usize),
13}
14
15#[derive(Debug)]
16pub struct PotentializedUnionFind<M: Group>
17where
18    M::Target: Eq,
19{
20    n: usize,
21    potential: RefCell<Vec<DiffOrSize<M::Target>>>,
22}
23
24impl<M: Group> PotentializedUnionFind<M>
25where
26    M::Target: Eq,
27{
28    pub fn new(size: usize) -> Self {
29        PotentializedUnionFind {
30            n: size,
31            potential: RefCell::new(vec![Size(1); size]),
32        }
33    }
34
35    /// xからみたyの差分を追加する  
36    /// つまり `potential[x] ◦ diff = potential[y]` となるようにする  
37    /// 今までの関係と矛盾しない場合、呼び出し前に差分が未定義なら`Ok(true)`、定義済みなら`Ok(false)`を返す  
38    /// 今までの関係と矛盾する場合、元々定義されているxから見たyの差分をeとして`Err(e)`を返す
39    pub fn relate(&mut self, x: usize, y: usize, diff: M::Target) -> Result<bool, M::Target> {
40        assert!(x < self.n);
41        assert!(y < self.n);
42        let (x, x_diff) = self.root_and_diff(x);
43        let (y, y_diff) = self.root_and_diff(y);
44        if x == y {
45            if M::binary_operation(&x_diff, &diff) == y_diff {
46                Ok(false)
47            } else {
48                Err(M::binary_operation(&M::inverse(&x_diff), &y_diff))
49            }
50        } else {
51            let mut pot = self.potential.borrow_mut();
52            if let (&Size(x_size), &Size(y_size)) = (&pot[x], &pot[y]) {
53                let x_root_to_y = M::binary_operation(&x_diff, &diff);
54                if x_size > y_size {
55                    let diff = M::binary_operation(&x_root_to_y, &M::inverse(&y_diff));
56                    pot[x] = Size(x_size + y_size);
57                    pot[y] = Diff(x, diff);
58                } else {
59                    let diff = M::binary_operation(&y_diff, &M::inverse(&x_root_to_y));
60                    pot[y] = Size(x_size + y_size);
61                    pot[x] = Diff(y, diff);
62                }
63                Ok(true)
64            } else {
65                unreachable!()
66            }
67        }
68    }
69
70    /// 代表元と、それから見た差分を求める  
71    /// つまり代表元を`root`として、`potential[root] ◦ diff = potential[x]` となるようなdiff
72    pub fn root_and_diff(&self, x: usize) -> (usize, M::Target) {
73        assert!(x < self.n);
74        let mut pot = self.potential.borrow_mut();
75        let mut buf = vec![];
76        let mut leader = x;
77        while let Diff(par, diff) = &pot[leader] {
78            buf.push((leader, diff.clone()));
79            leader = *par;
80        }
81        buf.push((leader, M::id_element()));
82        buf.reverse();
83        for i in 1..buf.len() {
84            let (v, ref diff) = buf[i];
85            let (par, _) = buf[i - 1];
86            let par_pot = if let Diff(_, par_pot) = &pot[par] {
87                par_pot
88            } else {
89                &buf[0].1
90            };
91            let new_diff = M::binary_operation(par_pot, diff);
92            pot[v] = Diff(leader, new_diff);
93        }
94        match pot[x] {
95            Diff(par, ref diff) => (par, diff.clone()),
96            Size(_) => (x, M::id_element()),
97        }
98    }
99
100    /// xから見たyの差分が定義されていれば返す  
101    /// つまり`potential[x] ◦ diff = potential[y]` となるようなdiff
102    pub fn diff(&self, x: usize, y: usize) -> Option<M::Target> {
103        assert!(x < self.n);
104        assert!(y < self.n);
105        let (x, x_diff) = self.root_and_diff(x);
106        let (y, y_diff) = self.root_and_diff(y);
107        if x == y {
108            Some(M::binary_operation(&M::inverse(&x_diff), &y_diff))
109        } else {
110            None
111        }
112    }
113
114    /// xとの差分が定義されているノードの数を返す
115    pub fn size(&self, x: usize) -> usize {
116        assert!(x < self.n);
117        let (x, _) = self.root_and_diff(x);
118        if let Size(size) = self.potential.borrow()[x] {
119            size
120        } else {
121            unreachable!()
122        }
123    }
124}
