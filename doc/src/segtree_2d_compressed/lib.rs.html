1//! Sparseな場合に対応するため、更新queryを先読みして座標圧縮する  
2//! もともと単位元で初期化されていると仮定する  
3//! 2次元なので可換性を要求  
4//! <https://drive.google.com/file/d/1bSjYiA-nSsHzBbCnLq1GeTpRzs2Ucm0q/view>で学びました  
5//! WaveletMatrixにSegmentTreeを載せたもののほうが速そうなので、基本これは使わないでよいかもしれないです
6
7use algebra::{Commutative, Monoid};
8use internal_type_traits::Integral;
9use segtree::SegTree;
10use std::ops::{Bound::*, Range, RangeBounds};
11
12/// Tは座標圧縮する型  
13#[derive(Debug)]
14pub struct SegTree2DCompressed<M: Monoid + Commutative, T: Integral> {
15    height_compressed: Vec<T>,
16    width_compressed: Vec<Vec<T>>,
17    data: Vec<SegTree<M>>,
18}
19
20impl<M: Monoid + Commutative, T: Integral> SegTree2DCompressed<M, T> {
21    /// オフラインで読み込んだ更新クエリのある座標を与え、単位元で初期化
22    pub fn new(update_queries: &[(T, T)]) -> Self {
23        let height_compressed = {
24            let mut tmp = update_queries.iter().map(|&(h, _)| h).collect::<Vec<_>>();
25            tmp.sort_unstable();
26            tmp.dedup();
27            tmp
28        };
29        let width_compressed = {
30            let mut tmp = vec![vec![]; height_compressed.len() * 2];
31            for &(h, w) in update_queries.iter() {
32                let h = height_compressed.binary_search(&h).unwrap() + height_compressed.len();
33                tmp[h].push(w);
34            }
35            for v in tmp.iter_mut() {
36                v.sort_unstable();
37                v.dedup();
38            }
39            for h in (1..height_compressed.len()).rev() {
40                let child_left = tmp[h * 2].clone();
41                let child_right = tmp[h * 2 + 1].clone();
42                tmp[h] = child_left.into_iter().chain(child_right).collect();
43                tmp[h].sort_unstable();
44                tmp[h].dedup();
45            }
46            tmp
47        };
48        let data = (0..height_compressed.len() * 2)
49            .map(|i| SegTree::<M>::new(width_compressed[i].len()))
50            .collect();
51        Self {
52            height_compressed,
53            width_compressed,
54            data,
55        }
56    }
57
58    pub fn get(&self, h: T, w: T) -> M::Target {
59        if let Ok(h) = self.height_compressed.binary_search(&h) {
60            let h = h + self.height_compressed.len();
61            if let Ok(w) = self.width_compressed[h].binary_search(&w) {
62                return self.data[h].get(w);
63            }
64        }
65        M::id_element()
66    }
67
68    /// 更新の都合上、setよりも二項演算の適用の方が定数倍速い  
69    pub fn add(&mut self, h: T, w: T, val: M::Target) {
70        let mut h = self
71            .height_compressed
72            .binary_search(&h)
73            .expect("h is not in update_queries");
74        h += self.height_compressed.len();
75        while h > 0 {
76            let cur_w_id = self.width_compressed[h]
77                .binary_search(&w)
78                .expect("w is not in update_queries");
79            let old_val = self.data[h].get(cur_w_id);
80            self.data[h].set(cur_w_id, M::binary_operation(&old_val, &val));
81            h >>= 1;
82        }
83    }
84
85    #[allow(clippy::collapsible_else_if, clippy::redundant_clone)]
86    pub fn set(&mut self, h: T, w: T, val: M::Target) {
87        // setよりもaddのような差分での更新の方が楽にかけるかも
88        let mut h = self
89            .height_compressed
90            .binary_search(&h)
91            .expect("h is not in update_queries");
92        h += self.height_compressed.len();
93        let mut pre_h = 2 * h;
94        let mut pre_val = val.clone();
95        while h > 0 {
96            let cur_w_id = self.width_compressed[h]
97                .binary_search(&w)
98                .expect("w is not in update_queries");
99            if h >= self.height_compressed.len() {
100                self.data[h].set(cur_w_id, val.clone());
101            } else {
102                let other_child = if pre_h == 2 * h {
103                    if let Ok(w) = self.width_compressed[2 * h + 1].binary_search(&w) {
104                        self.data[2 * h + 1].get(w)
105                    } else {
106                        M::id_element()
107                    }
108                } else {
109                    if let Ok(w) = self.width_compressed[2 * h].binary_search(&w) {
110                        self.data[2 * h].get(w)
111                    } else {
112                        M::id_element()
113                    }
114                };
115                let new_val = M::binary_operation(&pre_val, &other_child);
116                pre_val = new_val.clone();
117                self.data[h].set(cur_w_id, new_val);
118            }
119            pre_h = h;
120            h >>= 1;
121        }
122    }

124    pub fn prod<R1: RangeBounds<T>, R2: RangeBounds<T>>(
125        &self,
126        height_range: R1,
127        width_range: R2,
128    ) -> M::Target {
129        let height_left = match height_range.start_bound() {
130            Included(&l) => l,
131            Excluded(&l) => l + T::one(),
132            Unbounded => T::min_value(),
133        };
134        let height_right = match height_range.end_bound() {
135            Included(&r) => r + T::one(),
136            Excluded(&r) => r,
137            Unbounded => T::max_value(),
138        };
139        assert!(height_left <= height_right);
140        let mut height_left = self.height_compressed.partition_point(|&h| h < height_left);
141        let mut height_right = self
142            .height_compressed
143            .partition_point(|&h| h < height_right);
144        height_left += self.height_compressed.len();
145        height_right += self.height_compressed.len();
146        let mut ret = M::id_element();
147        while height_left < height_right {
148            if height_left & 1 != 0 {
149                let w_range = self.calc_row_range(height_left, &width_range);
150                ret = M::binary_operation(&ret, &self.data[height_left].prod(w_range));
151                height_left += 1;
152            }
153            if height_right & 1 != 0 {
154                height_right -= 1;
155                let w_range = self.calc_row_range(height_right, &width_range);
156                ret = M::binary_operation(&ret, &self.data[height_right].prod(w_range));
157            }
158            height_left >>= 1;
159            height_right >>= 1;
160        }
161        ret
162    }
163
164    fn calc_row_range<R1: RangeBounds<T>>(&self, h: usize, width_range: &R1) -> Range<usize> {
165        let w_left = match width_range.start_bound() {
166            Included(&l) => l,
167            Excluded(&l) => l + T::one(),
168            Unbounded => T::min_value(),
169        };
170        let w_right = match width_range.end_bound() {
171            Included(&r) => r + T::one(),
172            Excluded(&r) => r,
173            Unbounded => T::max_value(),
174        };
175        let w_left = self.width_compressed[h].partition_point(|&w| w < w_left);
176        let w_right = self.width_compressed[h].partition_point(|&w| w < w_right);
177        w_left..w_right
178    }
