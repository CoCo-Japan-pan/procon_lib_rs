3/// n以下のmod逆元列挙を`O(n)`で行う  
4/// index 0には便宜的に0を入れたn+1の長さの配列を返す
5pub fn enumerate_invs<M: ModInt>(n: usize) -> Vec<M> {
6    let n = n + 1;
7    assert!(n <= M::modulus() as usize);
8    let mut invs = vec![M::raw(0); n];
9    if n <= 1 {
10        return invs;
11    }
12    invs[1] = M::raw(1);
13    for i in 2..n {
14        invs[i] = -invs[M::modulus() as usize % i] * M::raw(M::modulus() / i as u32);
15    }
16    invs
17}
