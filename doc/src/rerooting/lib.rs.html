1//! 全方位木DP  
2//! 辺が向きつきで行きと帰りで異なる場合に対応しづらいので、ここでは頂点を用いて表している  
3//! 従って辺のコストとかは外でhashmap等で管理することになる  
4
5use algebra::{Commutative, Monoid};
6
7#[derive(Debug)]
8pub struct Rerooting<M: Monoid + Commutative, F: FnMut(&M::Target, usize, usize) -> M::Target> {
9    vertex_cnt: usize,
10    /// 根を0とした場合の各頂点を根とする部分木のDPテーブル
11    subtree_memo: Vec<M::Target>,
12    /// 各頂点を根とした木全体のDPテーブル
13    ans: Vec<M::Target>,
14    add_root: F,
15}
16
17impl<M: Monoid + Commutative, F: FnMut(&M::Target, usize, usize) -> M::Target> Rerooting<M, F> {
18    /// モノイド`M`は`add_root`によりできた「部分木+一辺」同士をmergeする関数を二項演算として持つ  
19    /// 葉にはモノイドの単位元が入る  
20    ///
21    /// `add_root(subtree: &M::Target, subtree_root: usize, new_root: usize) -> M::Target`  
22    /// `add_root`は部分木に頂点 `subtree_root → new_root` の辺を追加する関数  
23    ///
24    /// モノイドの型指定のために、`Rerooting::<Monoid, _>::new(..)`として下さい  
25    pub fn new(graph: &Vec<Vec<usize>>, add_root: F) -> Self {
26        let vertex_cnt = graph.len();
27        let subtree_memo = vec![M::id_element(); vertex_cnt];
28        let ans = vec![M::id_element(); vertex_cnt];
29        let mut ret = Self {
30            vertex_cnt,
31            subtree_memo,
32            ans,
33            add_root,
34        };
35        ret.dfs(graph, 0, usize::MAX);
36        ret.bfs(graph, 0, usize::MAX, M::id_element());
37        ret
38    }
39
40    pub fn get_ans(&self, root: usize) -> M::Target {
41        assert!(root < self.vertex_cnt);
42        self.ans[root].clone()
43    }
44
45    fn dfs(&mut self, graph: &Vec<Vec<usize>>, v: usize, p: usize) {
46        for &to in &graph[v] {
47            if to == p {
48                continue;
49            }
50            self.dfs(graph, to, v);
51            let memo = (self.add_root)(&self.subtree_memo[to], to, v);
52            self.subtree_memo[v] = M::binary_operation(&self.subtree_memo[v], &memo);
53        }
54    }
55
56    fn bfs(&mut self, graph: &Vec<Vec<usize>>, v: usize, p: usize, par_val: M::Target) {
57        // 左右から累積和を取っておく
58        let mut buf = Vec::with_capacity(graph[v].len());
59        for &to in &graph[v] {
60            if to == p {
61                continue;
62            } else {
63                buf.push((self.add_root)(&self.subtree_memo[to], to, v));
64            }
65        }
66        let mut left_sum = vec![M::id_element(); buf.len() + 1];
67        let mut right_sum = vec![M::id_element(); buf.len() + 1];
68        for i in 0..buf.len() {
69            left_sum[i + 1] = M::binary_operation(&left_sum[i], &buf[i]);
70        }
71        for i in (0..buf.len()).rev() {
72            right_sum[i] = M::binary_operation(&buf[i], &right_sum[i + 1]);
73        }
74        if p == usize::MAX {
75            self.ans[v] = left_sum.last().unwrap().clone();
76        } else {
77            self.ans[v] = M::binary_operation(left_sum.last().unwrap(), &par_val);
78        }
79
80        // 子に伝播
81        for (i, &to) in graph[v].iter().filter(|v| **v != p).enumerate() {
82            // 一つも部分木をmergeしないなら、leafを用いる
83            let propagate = M::binary_operation(
84                &par_val,
85                &M::binary_operation(&left_sum[i], &right_sum[i + 1]),
86            );
87            let par_val = (self.add_root)(&propagate, v, to);
88            self.bfs(graph, to, v, par_val);
89        }
90    }
91}
