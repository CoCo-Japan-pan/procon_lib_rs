1//! <https://nyaannyaan.github.io/library/segment-tree/segment-tree-beats-abstract.hpp>
2//! をもとにしています  
3//! 失敗せずに作用を適用できるものについてのみ`push`による遅延伝播を行い、失敗するものについてはボトムアップに計算する  
4//! 作用の失敗回数によい上界が存在するように設計する必要がある  
5//! 作用の成功部分を部分的に伝播させたいので、作用の合成は定義せず、
6//! 成功した作用の情報を載せたノードからその子ノードへの`push`を定義する  
7//! つまりノードは作用のうち成功している(=伝播可能な)部分的な情報をモノイドに加えて持つ  
8//! 作用については、`apply`で定義し、成功したら`true`、失敗したら`false`を返すようにする

13use std::ops::{Bound::*, RangeBounds};

15/// Segment Tree Beats における内部のノード  
16/// モノイド構造を持ちつつ、作用が成功した遅延情報も持つ
17/// (作用が適用されたモノイド自体から復元できるならば遅延情報は必要ない)  
18/// この遅延情報は、自身には適用済みだが、子孫には反映されていない(pushをして初めて子に伝播する)
19pub trait BeatsNode: Clone {
20    type Action;
21    fn id_node() -> Self;
22    fn binary_operation(lhs: &Self, rhs: &Self) -> Self;
23    /// 成功した作用の情報を載せたノードからその子ノードへの伝播し、そのノードの遅延を解消  
24    /// 特定のノードに成功する作用は、その子ノードでも成功するはず  
25    /// つまり tag/break condition のどちらかになるはずである
26    fn push(&mut self, child_node_left: &mut Self, child_node_right: &mut Self);
27    /// 作用の適用 成功したら`true`、失敗したら`false`を返す  
28    /// break condition (何も作用しない) や tag condition (モノイドを作用に基づき更新できる) なら`true`を返すということ  
29    /// 失敗したら子ノードにpushした上で同じ作用を適用するので、失敗する場合は特に変更しなくてよい  
30    /// 区間の長さ1に対しては必ず成功するはず
31    fn apply(&mut self, action: &Self::Action) -> bool;
32}
33
34#[derive(Debug)]
35pub struct LazySegtreeBeats<Node: BeatsNode> {
36    range_size: usize,
37    leaf_size: usize,
38    log: usize,
39    nodes: Vec<Node>,
40}
41
42impl<Node: BeatsNode> From<Vec<Node>> for LazySegtreeBeats<Node> {
43    fn from(mut v: Vec<Node>) -> Self {
44        let range_size = v.len();
45        let log = ceil_log2(range_size as u32) as usize;
46        let leaf_size = 1 << log;
47        let mut nodes = vec![Node::id_node(); leaf_size];
48        nodes.reserve(leaf_size);
49        nodes.append(&mut v);
50        nodes.append(&mut vec![Node::id_node(); leaf_size - range_size]);
51        let mut ret = Self {
52            range_size,
53            leaf_size,
54            log,
55            nodes,
56        };
57        for i in (1..leaf_size).rev() {
58            ret.update(i);
59        }
60        ret
61    }
62}
63
64impl<Node: BeatsNode> LazySegtreeBeats<Node> {
65    pub fn new(n: usize) -> Self {
66        vec![Node::id_node(); n].into()
67    }
68    pub fn prod<R: RangeBounds<usize>>(&mut self, range: R) -> Node {
69        let (mut l, mut r) = self.get_range(range);
70        if l == r {
71            return Node::id_node();
72        }
73        l += self.leaf_size;
74        r += self.leaf_size;
75        for i in (1..=self.log).rev() {
76            if ((l >> i) << i) != l {
77                self.push(l >> i);
78            }
79            if ((r >> i) << i) != r {
80                self.push((r - 1) >> i);
81            }
82        }
83
84        let mut sml = Node::id_node();
85        let mut smr = Node::id_node();
86        while l < r {
87            if l & 1 != 0 {
88                sml = Node::binary_operation(&sml, &self.nodes[l]);
89                l += 1;
90            }
91            if r & 1 != 0 {
92                r -= 1;
93                smr = Node::binary_operation(&self.nodes[r], &smr);
94            }
95            l >>= 1;
96            r >>= 1;
97        }
98        Node::binary_operation(&sml, &smr)
99    }
100
101    pub fn apply_range<R: RangeBounds<usize>>(&mut self, range: R, action: &Node::Action) {
102        let (mut l, mut r) = self.get_range(range);
103        if l == r {
104            return;
105        }
106        l += self.leaf_size;
107        r += self.leaf_size;
108        for i in (1..=self.log).rev() {
109            if ((l >> i) << i) != l {
110                self.push(l >> i);
111            }
112            if ((r >> i) << i) != r {
113                self.push((r - 1) >> i);
114            }
115        }
116        {
117            let l_copy = l;
118            let r_copy = r;
119            while l < r {
120                if l & 1 != 0 {
121                    self.apply(l, action);
122                    l += 1;
123                }
124                if r & 1 != 0 {
125                    r -= 1;
126                    self.apply(r, action);
127                }
128                l >>= 1;
129                r >>= 1;
130            }
131            l = l_copy;
132            r = r_copy;
133        }
134        for i in 1..=self.log {
135            if ((l >> i) << i) != l {
136                self.update(l >> i);
137            }
138            if ((r >> i) << i) != r {
139                self.update((r - 1) >> i);
140            }
141        }
142    }
143
144    fn get_range<R: RangeBounds<usize>>(&self, range: R) -> (usize, usize) {
145        let l = match range.start_bound() {
146            Included(&l) => l,
147            Excluded(&l) => l + 1,
148            Unbounded => 0,
149        };
150        let r = match range.end_bound() {
151            Included(&r) => r + 1,
152            Excluded(&r) => r,
153            Unbounded => self.range_size,
154        };
155        assert!(l <= r && r <= self.range_size);
156        (l, r)
157    }
158
159    fn push(&mut self, i: usize) {
160        let ptr = self.nodes.as_mut_ptr();
161        unsafe {
162            self.nodes[i].push(&mut *ptr.add(i << 1), &mut *ptr.add((i << 1) | 1));
163        }
164    }
165    fn update(&mut self, i: usize) {
166        self.nodes[i] = Node::binary_operation(&self.nodes[i << 1], &self.nodes[(i << 1) | 1]);
167    }
168    fn apply(&mut self, i: usize, action: &Node::Action) {
169        let res = self.nodes[i].apply(action);
170        // 作用が失敗したら子ノードに任せてボトムアップに計算
171        if (i < self.leaf_size) && !res {
172            self.push(i);
173            self.apply(i << 1, action);
174            self.apply((i << 1) | 1, action);
175            self.update(i);
176        }
177    }
178}
