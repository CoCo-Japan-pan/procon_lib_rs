1//! 動的セグメント木 必要なノードのみ確保する  
2//! 単位元で初期化する  
3//! <https://lorent-kyopro.hatenablog.com/entry/2021/03/12/025644>
4
5use algebra::Monoid;
6use std::ops::RangeBounds;
7
8#[derive(Debug)]
9struct Node<M: Monoid> {
10    left_child: Option<Box<Node<M>>>,
11    right_child: Option<Box<Node<M>>>,
12    index: usize,
13    value: M::Target,
14    product: M::Target,
15}
16
17impl<M: Monoid> Node<M> {
18    fn new(index: usize, value: M::Target) -> Self {
19        Self {
20            left_child: None,
21            right_child: None,
22            index,
23            product: value.clone(),
24            value,
25        }
26    }
27
28    fn update(&mut self) {
29        let id_element = M::id_element();
30        let left_value = self
31            .left_child
32            .as_ref()
33            .map_or(&id_element, |node| &node.product);
34        let right_value = self
35            .right_child
36            .as_ref()
37            .map_or(&id_element, |node| &node.product);
38        self.product =
39            M::binary_operation(&M::binary_operation(left_value, &self.value), right_value);
40    }
41}
42
43type NodePtr<M> = Option<Box<Node<M>>>;
44
45fn set_node<M: Monoid>(
46    node: &mut NodePtr<M>,
47    mut index: usize,
48    mut value: M::Target,
49    left: usize,
50    right: usize,
51) {
52    if node.is_none() {
53        // 余計に潜らず、メモリ確保を高々1回に抑える
54        *node = Some(Box::new(Node::new(index, value)));
55        return;
56    }
57    let node = node.as_mut().unwrap();
58    if node.index == index {
59        node.value = value;
60        node.update();
61        return;
62    }
63    let half = (left + right) >> 1;
64    if index < half {
65        // 左に行くほどindexが小さくなる条件を満たすようにする
66        if index > node.index {
67            std::mem::swap(&mut node.value, &mut value);
68            std::mem::swap(&mut node.index, &mut index);
69        }
70        set_node(&mut node.left_child, index, value, left, half);
71    } else {
72        // 右に行くほどindexが大きくなる条件を満たすようにする
73        if index < node.index {
74            std::mem::swap(&mut node.value, &mut value);
75            std::mem::swap(&mut node.index, &mut index);
76        }
77        set_node(&mut node.right_child, index, value, half, right);
78    }
79    node.update();
80}
81
82fn get_node<M: Monoid>(node: &NodePtr<M>, index: usize, left: usize, right: usize) -> M::Target {
83    if node.is_none() {
84        return M::id_element();
85    }
86    let node = node.as_ref().unwrap();
87    if node.index == index {
88        return node.value.clone();
89    }
90    let half = (left + right) >> 1;
91    if index < half {
92        get_node(&node.left_child, index, left, half)
93    } else {
94        get_node(&node.right_child, index, half, right)
95    }
96}
97
98fn prod_node<M: Monoid>(
99    node: &NodePtr<M>,
100    prod_left: usize,
101    prod_right: usize,
102    left: usize,
103    right: usize,
104) -> M::Target {
105    if node.is_none() || right <= prod_left || prod_right <= left {
106        return M::id_element();
107    }
108    let node = node.as_ref().unwrap();
109    if prod_left <= left && right <= prod_right {
110        return node.product.clone();
111    }
112    let half = (left + right) >> 1;
113    let mut res = M::id_element();
114    res = M::binary_operation(
115        &res,
116        &prod_node(&node.left_child, prod_left, prod_right, left, half),
117    );
118    if prod_left <= node.index && node.index < prod_right {
119        res = M::binary_operation(&res, &node.value);
120    }
121    res = M::binary_operation(
122        &res,
123        &prod_node(&node.right_child, prod_left, prod_right, half, right),
124    );
125    res
126}

128#[derive(Debug)]
129pub struct DynamicSegTree<M: Monoid> {
130    range_size: usize,
131    root_node: NodePtr<M>,
132}
133
134impl<M: Monoid> DynamicSegTree<M> {
135    /// range_size個の単位元で初期化
136    pub fn new(range_size: usize) -> Self {
137        Self {
138            range_size,
139            root_node: None,
140        }
141    }
142
143    pub fn set(&mut self, index: usize, value: M::Target) {
144        assert!(index < self.range_size);
145        set_node(&mut self.root_node, index, value, 0, self.range_size);
146    }
147
148    pub fn get(&self, index: usize) -> M::Target {
149        assert!(index < self.range_size);
150        get_node(&self.root_node, index, 0, self.range_size)
151    }
152
153    pub fn all_prod(&self) -> M::Target {
154        self.root_node
155            .as_ref()
156            .map_or(M::id_element(), |node| node.product.clone())
157    }
158
159    pub fn prod<R: RangeBounds<usize>>(&self, range: R) -> M::Target {
160        use std::ops::Bound::*;
161        let start = match range.start_bound() {
162            Included(&l) => l,
163            Excluded(&l) => l + 1,
164            Unbounded => 0,
165        };
166        let end = match range.end_bound() {
167            Included(&r) => r + 1,
168            Excluded(&r) => r,
169            Unbounded => self.range_size,
170        };
171        assert!(start <= end && end <= self.range_size);
172        if start == 0 && end == self.range_size {
173            return self.all_prod();
174        }
175        prod_node(&self.root_node, start, end, 0, self.range_size)
176    }
177}
